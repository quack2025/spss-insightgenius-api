5-Minute Quickstart

1. Install

npm install insightgenius-sdk

2. Initialize

import { InsightGenius } from "insightgenius-sdk";

const ig = new InsightGenius("sk_test_your_key");

3. Upload

const file = new Blob([readFileSync("survey.sav")]);
const { file_id } = await ig.upload(file, "survey.sav");
// file_id valid for 30 minutes

4. Analyze

const ct = await ig.crosstab(file_id, {
  row: "Q1_Satisfaction",
  col: "Gender",
  significance_level: 0.95,
});
// ct.significant_pairs → [{ row: "5", col1: "1.0", col2: "2.0", letter: "B" }]
// Column A is significantly higher than Column B

5. Export

const excel = await ig.tabulate(file_id, {
  banners: ["Gender", "AgeGroup"],
  stubs: ["_all_"],
  significance_level: 0.95,
  include_means: true,
});
writeFileSync("report.xlsx", Buffer.from(excel));
